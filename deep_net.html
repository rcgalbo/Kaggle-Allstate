# define base model
def baseline_model():
    # create model
    model = Sequential()
    model.add(Dense(350 , input_dim=246 , init='normal', activation='relu'))
    model.add(Dense(1, init='normal'))
    # Compile model
    model.compile(loss='mean_absolute_error', optimizer='adam')
    return model
